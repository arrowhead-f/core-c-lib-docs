9// An example of how to construct a JSON representation using C99 standard
10// library functions and our ah_json_str_escape() function.
11
12#include <ah/err.h>
13#include <ah/json.h>
14#include <limits.h>
15#include <stddef.h>
16#include <stdio.h>
17#include <stdlib.h>
18#include <string.h>
19
20int main(void)
21{
22 ah_err_t err;
23 char err_buf[128u];
24 const char* err_ctx;
25
26 const char* sensor_id = "aa-xx-142b";
27 float kelvin = 296.55f;
28
29 // Here, we assume that `sensor_id` may contain characters that must be
30 // escaped before they can be part of a JSON message. If we had known that
31 // `sensor_id` will never contain control characters, backslashes or double
32 // quotes, we could have skipped escaping it.
33
34 char sensor_id_buf[32u];
35 size_t sensor_id_buf_length = sizeof(sensor_id_buf);
36 err = ah_json_str_escape(sensor_id, strlen(sensor_id), sensor_id_buf, &sensor_id_buf_length);
37 if (err != AH_ENONE) {
38 err_ctx = "ah_json_str_escape()";
39 goto handle_err;
40 }
41 if (sensor_id_buf_length > INT_MAX) {
42 err = AH_EDOM;
43 err_ctx = "ah_json_str_escape()";
44 goto handle_err;
45 }
46
47 // We are not ready to produce the actual JSON object.
48
49 char out_buf[128u];
50 int size = snprintf(out_buf, sizeof(out_buf), "{\"sensor-id\":\"%.*s\",\"kelvin\":%f}",
51 (int) sensor_id_buf_length, sensor_id_buf, kelvin);
52
53 if (size < 0) {
54 perror(NULL);
55 exit(EXIT_FAILURE);
56 }
57
58 if (((size_t) size) > sizeof(out_buf)) {
59 fputs("out_buf overflowed\n", stderr);
60 exit(EXIT_FAILURE);
61 }
62
63 // `out_buf` now contains a JSON object of size `res`. You may, for example,
64 // choose to send the contents of `out_buf` over the Internet or some other
65 // network. We print it just to make it visible if you run this program.
66
67 printf("%.*s\n", size, out_buf); // Prints `{"sensor_name":"aa-xx-142b","kelvin":296.55}`.
68
69 exit(EXIT_SUCCESS);
70
71handle_err:
72 ah_strerror_r(err, err_buf, sizeof(err_buf));
73 fprintf(stderr, "%s failed: %s\n", err_ctx, err_buf);
74 exit(EXIT_FAILURE);
75}
