9// An example of how to interpret a JSON representation using our
10// ah_json_parse() function and some relevant utility functions.
11
12#include <ah/err.h>
13#include <ah/json.h>
14#include <stdbool.h>
15#include <stdio.h>
16#include <stdlib.h>
17#include <string.h>
18
19int main(void)
20{
21 ah_err_t err;
22 char err_buf[128u];
23 const char* err_ctx;
24
25 // The JSON text we are interpreting.
26 const char* json = "{\"sensor-id\":\"aa-xx-142b\",\"kelvin\":296.549988}";
27 ah_buf_t src = ah_buf_from((uint8_t*) json, (uint32_t) strlen(json));
28
29 // Parsing the JSON text populates `buf`.
30 ah_json_buf_t buf = { 0u };
31 err = ah_json_parse(src, &buf);
32 if (err != AH_ENONE) {
33 err_ctx = "ah_json_parse()";
34 goto handle_err;
35 }
36
37 // We use these to ensure the object we expect contains all mandatory fields.
38 bool has_sensor_id = false;
39 bool has_kelvin = false;
40
41 // We use these to collect the data in the object.
42 char sensor_id[16u] = { 0u };
43 float kelvin = -1.0f;
44
45 // We begin validation by ensuring that a root value of the object type exists in `buf`.
46 if (buf.length == 0u || buf.values[0u].type != AH_JSON_TYPE_OBJECT) {
47 err = AH_EINVAL;
48 err_ctx = "'expecting object at root'";
49 goto handle_err;
50 }
51
52 // We then iterate through the children of the object.
53 for (size_t i = 1u; i < buf.length; i += 1u) {
54 ah_json_val_t* val = &buf.values[i];
55
56 // As long as the level is unchanged, the current value is going to be a
57 // string and the next value is guaranteed to have the same level. Why?
58 // Because we know that there is an object at level 0u, and objects are
59 // guaranteed to always have even numbers of child values where even
60 // children (beginning with the 0th child) is a key of type string.
61 if (val->level != 1u) {
62 continue;
63 }
64
65 // Is the key string equal to "sensor-id"? If so, make sure we have not
66 // seen a sensor-id before and that the next value is of type string.
67 // Save that value to the `sensor_id` variable. To improve performance,
68 // you may choose to use memcmp() directly instead of
69 // ah_json_str_compare(). This requires the sender not to use any escape
70 // sequences in this particular key, however.
71 if (ah_json_str_compare("sensor-id", 9u, val->base, val->length) == 0u) {
72 if (has_sensor_id) {
73 err = AH_EDUP;
74 err_ctx = "ah_json_str_compare()";
75 goto handle_err;
76 }
77
78 i += 1u;
79
80 val = &buf.values[i];
81 if (val->type != AH_JSON_TYPE_STRING) {
82 err = AH_EINVAL;
83 err_ctx = "'expecting \"sensor-id\" to have string value'";
84 goto handle_err;
85 }
86
87 size_t sensor_id_length = sizeof(sensor_id);
88 err = ah_json_str_unescape(val->base, val->length, sensor_id, &sensor_id_length);
89 if (err != AH_ENONE) {
90 err_ctx = "ah_json_str_unescape()";
91 goto handle_err;
92 }
93 if (sensor_id_length == sizeof(sensor_id)) {
94 sensor_id_length -= 1u;
95 }
96 sensor_id[sensor_id_length] = '\0';
97
98 has_sensor_id = true;
99 continue;
100 }
101
102 // Is the key string equal to "kelvin"? If so, make sure we have not
103 // seen a kelvin key before and that the next value is of type number.
104 // Parse that value to the `kelvin` variable.
105 if (ah_json_str_compare("kelvin", 6u, val->base, val->length) == 0u) {
106 if (has_kelvin) {
107 err = AH_EDUP;
108 err_ctx = "ah_json_str_compare()";
109 goto handle_err;
110 }
111
112 i += 1u;
113
114 val = &buf.values[i];
115 if (val->type != AH_JSON_TYPE_NUMBER) {
116 err = AH_EINVAL;
117 err_ctx = "'expecting \"kelvin\" to have number value'";
118 goto handle_err;
119 }
120
121 errno = 0;
122 kelvin = strtof(val->base, NULL);
123 if (errno != 0) {
124 err = errno;
125 err_ctx = "strtof";
126 goto handle_err;
127 }
128
129 has_kelvin = true;
130 continue;
131 }
132 }
133
134 // Make sure we have seen all mandatory fields.
135
136 if (!has_sensor_id) {
137 err = AH_EINVAL;
138 err_ctx = "'expecting \"sensor_id\" in object'";
139 goto handle_err;
140 }
141
142 if (!has_kelvin) {
143 err = AH_EINVAL;
144 err_ctx = "'expecting \"kelvin\" in object'";
145 goto handle_err;
146 }
147
148 // Print the result, if everything worked out as expected.
149
150 printf("Interpretation result:\n"
151 "\tsensor_id: %15s\n"
152 "\tkelvin: %15f\n",
153 sensor_id, kelvin);
154
155 exit(EXIT_SUCCESS);
156
157handle_err:
158 ah_strerror_r(err, err_buf, sizeof(err_buf));
159 fprintf(stderr, "%s failed: %s\n", err_ctx, err_buf);
160 exit(EXIT_FAILURE);
161}
